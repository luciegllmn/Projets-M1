K = range(1, 6)
score_knn = []
for i in K:
    knn = KNeighborsRegressor(n_neighbors = i)
    knn.fit(X_train, y_train)
    le_score = knn.score(X_test, y_test)
    score_knn.append(le_score*100)

sb.lineplot(K, score_knn)
